CMAKE_MINIMUM_REQUIRED(VERSION 2.8)

PROJECT(OclMD)
SET(OCLMD_LIBRARY_NAME OclMD)
SET(SHARED_TARGET ${OCLMD_LIBRARY_NAME})
SET(STATIC_TARGET ${OCLMD_LIBRARY_NAME}_static)

SET(CMAKE_BUILD_TYPE Debug)
SET(OCLMD_SRC_SUBDIRS "src")

INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR}/include)
# start compiling each source file

SET(FILENAME_LIST "")
FILE(GLOB SRC_PROGS "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp")

FOREACH(SRC_PROG ${SRC_PROGS})
    GET_FILENAME_COMPONENT(FILENAME ${SRC_PROG} NAME)
    SET(FILENAME_LIST ${FILENAME_LIST}  src/${FILENAME})
ENDFOREACH(SRC_PROG ${SRC_PROGS})
MESSAGE(${FILENAME_LIST})

ADD_LIBRARY(${SHARED_TARGET} SHARED ${FILENAME_LIST})


SUBDIRS(tests)
ENABLE_TESTING()
